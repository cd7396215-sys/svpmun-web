  document.getElementById('nav-bar').classList.toggle('scrolled', window.scrollY > 40);

/* NAV TOGGLE */
var navOpen = false;
function toggleNav(){
  navOpen = !navOpen;
  document.getElementById('nav-bar').classList.toggle('menu-open', navOpen);
  document.getElementById('nav-overlay').classList.toggle('open', navOpen);
  document.getElementById('nbIcon').classList.toggle('open', navOpen);
  document.body.style.overflow = navOpen ? 'hidden' : '';
}
document.addEventListener('keydown', function(e){ if(e.key === 'Escape' && navOpen) toggleNav(); });

/* LOADER */
(function(){
  var lines    = ['dl0','dl1','dl2','dl3','dl4'];
  var fill     = document.getElementById('ldProgress');
  var pct      = document.getElementById('ldPct');
  var ag       = document.getElementById('access-granted');
  var loader   = document.getElementById('loader');
  var progress = 0;
  var lineIdx  = 0;

  function showNext(){
    if(lineIdx >= lines.length) return;
    document.getElementById(lines[lineIdx]).classList.add('active');
    lineIdx++;
    if(lineIdx < lines.length) setTimeout(showNext, 300 + Math.random()*200);
  }
  setTimeout(showNext, 1200);

  var iv = setInterval(function(){
    progress = Math.min(100, progress + Math.random()*6 + 1);
    fill.style.width = progress + '%';
    pct.textContent  = Math.floor(progress) + '%';
    if(progress >= 100){
      clearInterval(iv);
      setTimeout(function(){
        ag.classList.add('show');
        setTimeout(function(){
          loader.classList.add('exit');
          setTimeout(function(){ loader.remove(); }, 950);
        }, 900);
      }, 400);
    }
  }, 50);
})();

  });
}, { threshold: 0.1 });
document.querySelectorAll('.rv, .divider').forEach(function(el){ io.observe(el); });
